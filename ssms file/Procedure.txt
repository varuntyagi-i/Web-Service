Create Proc spGetDataByID
@ID nvarchar(50)
as
Begin
 Select INSERT_DATE,INSERT_USER,PRODUCED_BY,SDAT_LABEL,
		UPDATE_DATE,UPDATE_USER,EXISTENCE_KIND,GUID,ID, 
		VERSION,NAME,ORIGINAL_SOURCE,REMARKS,SOURCE,BASIN_ID,
		BOUNDARY_AREA_ID,DISCOVERY_DATE,DISCOVERY_TYPE,
		DISCOVERY_WELL_ID,FIELD_TYPE,LOCATION_ID,
		PARENT_FIELD_ID,PROJECT
 from FIELD where ID = @ID
End
